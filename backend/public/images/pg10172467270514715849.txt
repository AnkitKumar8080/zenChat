import cv2

image = cv2.imread("PHOTO.jpg")

gaussianBlur = cv2.GaussianBlur(image, (15, 15), 0)

mediaBlur = cv2.medianBlur(image, 15)

bilateralFilter =  cv2.bilateralFilter(image, 15, 75, 75)

cv2.imshow("gaussian blur", gaussianBlur)
cv2.imshow("median blur ", mediaBlur)
cv2.imshow("bilateral filter ", bilateralFilter)

cv2.waitKey(0)
cv2.destroyAllWindows()
